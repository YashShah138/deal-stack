    ```typescript
    import { SupabaseClient } from '@supabase/supabase-js';
    import { BaseProvider, RateLimitExceededError } from '../types';
    import {
      RentcastPropertySchema,
      RentcastRentEstimateSchema,
      RentcastValueEstimateSchema,
      RentcastMarketTrendsSchema,
    } from './schemas/rentcast';

    const RENTCAST_BASE_URL = 'https://api.rentcast.io/v1';
    const MONTHLY_CALL_LIMIT = 50;

    const ENDPOINT_SCHEMAS: Record<string, z.ZodSchema> = {
      'properties': RentcastPropertySchema,
      'avm/rent/long-term': RentcastRentEstimateSchema,
      'avm/value': RentcastValueEstimateSchema,
      'markets': RentcastMarketTrendsSchema,
    };

    export class RentcastProvider implements BaseProvider {
      readonly name = 'rentcast';
      readonly defaultTtlDays = 30;

      async fetch<T>(endpoint: string, params: Record<string, string>): Promise<T> {
        const apiKey = process.env.RENTCAST_API_KEY;
        if (!apiKey) throw new Error('RENTCAST_API_KEY environment variable is not set');

        const url = new URL(`${RENTCAST_BASE_URL}/${endpoint}`);
        for (const [key, value] of Object.entries(params)) {
          url.searchParams.set(key, value);
        }

        const response = await fetch(url.toString(), {
          headers: { 'X-Api-Key': apiKey },
        });

        if (!response.ok) {
          throw new Error(`Rentcast API error: ${response.status} ${response.statusText}`);
        }

        const json = await response.json();
        const schema = ENDPOINT_SCHEMAS[endpoint];
        if (schema) {
          return schema.parse(json) as T;
        }
        return json as T;
      }

      async checkRateLimit(supabase: SupabaseClient): Promise<void> {
        const count = await this.getMonthlyCallCount(supabase);
        if (count >= MONTHLY_CALL_LIMIT) {
          throw new RateLimitExceededError('rentcast', MONTHLY_CALL_LIMIT);
        }
      }

      private async getMonthlyCallCount(supabase: SupabaseClient): Promise<number> {
        const monthStart = new Date();
        monthStart.setDate(1);
        monthStart.setHours(0, 0, 0, 0);

        const { count } = await supabase
          .from('api_usage_log')
          .select('*', { count: 'exact', head: true })
          .eq('provider', 'rentcast')
          .eq('cache_hit', false)
          .gte('called_at', monthStart.toISOString());

        return count ?? 0;
      }
    }

    // Convenience function for UI display (DATA-04)
    export async function getRentcastUsage(supabase: SupabaseClient): Promise<{ used: number; limit: number }> {
      const provider = new RentcastProvider();
      const monthStart = new Date();
      monthStart.setDate(1);
      monthStart.setHours(0, 0, 0, 0);

      const { count } = await supabase
        .from('api_usage_log')
        .select('*', { count: 'exact', head: true })
        .eq('provider', 'rentcast')
        .eq('cache_hit', false)
        .gte('called_at', monthStart.toISOString());

      return { used: count ?? 0, limit: MONTHLY_CALL_LIMIT };
    }
    ```

    Note: Import `z` from 'zod' in the provider file for the ENDPOINT_SCHEMAS typing. The Zod schemas validate responses before they get cached -- corrupt data in cache propagates errors indefinitely (per RESEARCH.md anti-patterns).
